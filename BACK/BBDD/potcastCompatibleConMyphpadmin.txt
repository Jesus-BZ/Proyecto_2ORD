-- *** Script SQL para MySQL (solo cambios mínimos) ***

-- 1. Eliminar tablas (recomendado para evitar errores al re-ejecutar)
DROP TABLE IF EXISTS comentarios;
DROP TABLE IF EXISTS potcast;
DROP TABLE IF EXISTS usuario;

-- 2. Crear las tablas
-- Cambios: NUMBER -> INT, VARCHAR2 -> VARCHAR, GENERATED ALWAYS AS IDENTITY -> AUTO_INCREMENT.
CREATE TABLE usuario (
    id_usuario INT PRIMARY KEY AUTO_INCREMENT,
    nombre VARCHAR(30),
    email VARCHAR(30)
);

-- Cambios: NUMBER -> INT, VARCHAR2 -> VARCHAR, GENERATED ALWAYS AS IDENTITY -> AUTO_INCREMENT.
CREATE TABLE potcast (
    id_potcast INT PRIMARY KEY AUTO_INCREMENT,
    titulo VARCHAR(30),
    valoracion DECIMAL(4,2) -- NUMBER(4,2) se convierte a DECIMAL(4,2)
);

-- Cambios: NUMBER -> INT, VARCHAR2 -> VARCHAR, GENERATED ALWAYS AS IDENTITY -> AUTO_INCREMENT.
-- Nota: La columna 'id_podcast' en 'comentarios' debe coincidir con el nombre de la clave foránea.
CREATE TABLE comentarios (
    id_valoracion INT PRIMARY KEY AUTO_INCREMENT,
    id_usuario INT NOT NULL,
    id_podcast INT NOT NULL, -- 'id_podcast' es el nombre de la columna en esta tabla
    valoracion DECIMAL(4,2), -- NUMBER(4,2) se convierte a DECIMAL(4,2)
    descripcion VARCHAR(30), -- 'descripcion' se mantiene como VARCHAR(30)
    FOREIGN KEY (id_usuario) REFERENCES usuario(id_usuario) ON DELETE CASCADE,
    FOREIGN KEY (id_podcast) REFERENCES potcast(id_potcast) ON DELETE CASCADE -- 'id_potcast' en la tabla 'potcast'
);
